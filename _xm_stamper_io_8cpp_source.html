214 void iWriteTinToFile(std::ofstream &a_file, const std::string &a_cardName, const BSHP<const TrTin> &a_tin)
215 {
216  XM_ENSURE_TRUE(a_file.is_open());
217  a_file << a_cardName + "\n";
218  const VecPt3d& points = a_tin->Points();
219  iWriteVecPt3dToFile(a_file, "POINTS", points);
220 
221  VecInt vTri = a_tin->Triangles();
222  std::vector<std::array<int, 3>> sortableTris;
223  // sort the triangles so that the point with the lowest index is first and then sort all
224  // triangles
225  for (size_t i = 0; i < vTri.size(); i += 3)
226  {
227  std::array<int, 3> tri = { -1, -1, -1 };
228  tri[0] = vTri[i + 0];
229  tri[1] = vTri[i + 1];
230  tri[2] = vTri[i + 2];
231  auto minIter = tri.begin();
232  if (tri[1] < tri[0] && tri[1] < tri[2])
233  minIter += 1;
234  else if (tri[2] < tri[0] && tri[2] < tri[1])
235  minIter += 2;
236  std::rotate(tri.begin(), minIter, tri.begin() + 3);
237  sortableTris.push_back(tri);
238  }
239  std::sort(sortableTris.begin(), sortableTris.end());
240  auto it = sortableTris.begin();
241  for (size_t i = 0; i < vTri.size(); i += 3, ++it)
242  {
243  vTri[i + 0] = (*it)[0];
244  vTri[i + 1] = (*it)[1];
245  vTri[i + 2] = (*it)[2];
246  }
247  a_file << "TRIANGLES " << vTri.size() << "\n";
248  for (size_t i = 0; i < vTri.size(); i += 3)
249  {
250  a_file << std::setw(10) << vTri[i + 0] << " "
251  << std::setw(10) << vTri[i + 1] << " "
252  << std::setw(10) << vTri[i + 2] << "\n";
253  }
254  //const VecInt2d& trisAdjToPts = a_tin->TrisAdjToPts();
255  a_file << "TRIS_ADJ_TO_PTS 0\n";
256  //iWriteVecInt2dToFile(a_file, "TRIS_ADJ_TO_PTS", trisAdjToPts);
257 } // iWriteTinToFile

662 bool XmStamperIo::ReadFromFile(std::ifstream &a_file)
663 {
664  XM_ENSURE_TRUE(a_file.is_open(), false);
665  std::string card;
666  while (a_file >> card)
667  {
668  if (stEqualNoCase(card, "CENTER_LINE"))
669  {
670  XM_ENSURE_TRUE(iReadVecPt3dFromFile(a_file, m_centerLine), false);
671  }
672  else if (stEqualNoCase(card, "STAMPING_TYPE"))
673  {
674  XM_ENSURE_TRUE(a_file >> m_stampingType, false);
675  }
676  else if (stEqualNoCase(card, "CROSS_SECTIONS"))
677  {
678  int numCs(0);
679  XM_ENSURE_TRUE(a_file >> numCs, false);
680  m_cs.assign(numCs, XmStampCrossSection());
681  for (auto &cs : m_cs)
682  {
683  XM_ENSURE_TRUE(cs.ReadFromFile(a_file), false);
684  }
685  }
686  else if (stEqualNoCase(card, "FIRST_END_CAP"))
687  {
688  XM_ENSURE_TRUE(m_firstEndCap.ReadFromFile(a_file), false);
689  }
690  else if (stEqualNoCase(card, "LAST_END_CAP"))
691  {
692  XM_ENSURE_TRUE(m_lastEndCap.ReadFromFile(a_file), false);
693  }
694  else if (stEqualNoCase(card, "BATHYMETRY"))
695  {
696  m_bathymetry = TrTin::New();
697  XM_ENSURE_TRUE(iReadTinFromFile(a_file, m_bathymetry), false);
698  }
699  else if (stEqualNoCase(card, "OUT_TIN"))
700  {
701  m_outTin = TrTin::New();
702  XM_ENSURE_TRUE(iReadTinFromFile(a_file, m_outTin), false);
703  }
704  else if (stEqualNoCase(card, "OUT_BREAK_LINES"))
705  {
706  XM_ENSURE_TRUE(iReadVecInt2dFromFile(a_file, m_outBreakLines), false);
707  }
708  else if (stEqualNoCase(card, "RASTER"))
709  {
710  XM_ENSURE_TRUE(m_raster.ReadFromFile(a_file), false);
711  }
712  else
713  {
714  XM_ASSERT(0);
715  }
716  }
717  return true;
718 } // XmStamperIo::ReadFromFile

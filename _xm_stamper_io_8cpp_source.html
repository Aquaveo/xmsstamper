215 void iWriteTinToFile(std::ofstream &a_file, const std::string &a_cardName, const BSHP<const TrTin> &a_tin)
216 {
217  XM_ENSURE_TRUE(a_file.is_open());
218  a_file << a_cardName + "\n";
219  const VecPt3d& points = a_tin->Points();
220  iWriteVecPt3dToFile(a_file, "POINTS", points);
221 
222  VecInt vTri = a_tin->Triangles();
223  std::vector<std::array<int, 3>> sortableTris;
224  // sort the triangles so that the point with the lowest index is first and then sort all
225  // triangles
226  for (size_t i = 0; i < vTri.size(); i += 3)
227  {
228  std::array<int, 3> tri = { -1, -1, -1 };
229  tri[0] = vTri[i + 0];
230  tri[1] = vTri[i + 1];
231  tri[2] = vTri[i + 2];
232  auto minIter = tri.begin();
233  if (tri[1] < tri[0] && tri[1] < tri[2])
234  minIter += 1;
235  else if (tri[2] < tri[0] && tri[2] < tri[1])
236  minIter += 2;
237  std::rotate(tri.begin(), minIter, tri.begin() + 3);
238  sortableTris.push_back(tri);
239  }
240  std::sort(sortableTris.begin(), sortableTris.end());
241  auto it = sortableTris.begin();
242  for (size_t i = 0; i < vTri.size(); i += 3, ++it)
243  {
244  vTri[i + 0] = (*it)[0];
245  vTri[i + 1] = (*it)[1];
246  vTri[i + 2] = (*it)[2];
247  }
248  a_file << "TRIANGLES " << vTri.size() << "\n";
249  for (size_t i = 0; i < vTri.size(); i += 3)
250  {
251  a_file << std::setw(10) << vTri[i + 0] << " "
252  << std::setw(10) << vTri[i + 1] << " "
253  << std::setw(10) << vTri[i + 2] << "\n";
254  }
255  //const VecInt2d& trisAdjToPts = a_tin->TrisAdjToPts();
256  a_file << "TRIS_ADJ_TO_PTS 0\n";
257  //iWriteVecInt2dToFile(a_file, "TRIS_ADJ_TO_PTS", trisAdjToPts);
258 } // iWriteTinToFile

663 bool XmStamperIo::ReadFromFile(std::ifstream &a_file)
664 {
665  XM_ENSURE_TRUE(a_file.is_open(), false);
666  std::string card;
667  while (a_file >> card)
668  {
669  if (stEqualNoCase(card, "CENTER_LINE"))
670  {
671  XM_ENSURE_TRUE(iReadVecPt3dFromFile(a_file, m_centerLine), false);
672  }
673  else if (stEqualNoCase(card, "STAMPING_TYPE"))
674  {
675  XM_ENSURE_TRUE(a_file >> m_stampingType, false);
676  }
677  else if (stEqualNoCase(card, "CROSS_SECTIONS"))
678  {
679  int numCs(0);
680  XM_ENSURE_TRUE(a_file >> numCs, false);
681  m_cs.assign(numCs, XmStampCrossSection());
682  for (auto &cs : m_cs)
683  {
684  XM_ENSURE_TRUE(cs.ReadFromFile(a_file), false);
685  }
686  }
687  else if (stEqualNoCase(card, "FIRST_END_CAP"))
688  {
689  XM_ENSURE_TRUE(m_firstEndCap.ReadFromFile(a_file), false);
690  }
691  else if (stEqualNoCase(card, "LAST_END_CAP"))
692  {
693  XM_ENSURE_TRUE(m_lastEndCap.ReadFromFile(a_file), false);
694  }
695  else if (stEqualNoCase(card, "BATHYMETRY"))
696  {
697  m_bathymetry = TrTin::New();
698  XM_ENSURE_TRUE(iReadTinFromFile(a_file, m_bathymetry), false);
699  }
700  else if (stEqualNoCase(card, "OUT_TIN"))
701  {
702  m_outTin = TrTin::New();
703  XM_ENSURE_TRUE(iReadTinFromFile(a_file, m_outTin), false);
704  }
705  else if (stEqualNoCase(card, "OUT_BREAK_LINES"))
706  {
707  XM_ENSURE_TRUE(iReadVecInt2dFromFile(a_file, m_outBreakLines), false);
708  }
709  else if (stEqualNoCase(card, "RASTER"))
710  {
711  XM_ENSURE_TRUE(m_raster.ReadFromFile(a_file), false);
712  }
713  else
714  {
715  XM_ASSERT(0);
716  }
717  }
718  return true;
719 } // XmStamperIo::ReadFromFile

103 void XmStampInterpCrossSectionImpl::InterpMissingCrossSections(XmStamperIo& a_)
104 {
105  m_io = &a_;
106 
107  // make sure shoulder index is not 0 on any cross section
108  for (size_t i = 0; i < m_io->m_cs.size(); ++i)
109  {
110  auto& cs(m_io->m_cs[i]);
111  if (cs.m_left.size() > 1 || cs.m_right.size() > 1)
112  {
113  if (cs.m_idxLeftShoulder < 1)
114  cs.m_idxLeftShoulder = 1;
115  if (cs.m_idxRightShoulder < 1)
116  cs.m_idxRightShoulder = 1;
117  }
118  }
119 
120  int first = FindFirstValidCrossSection();
121  // copy this cross section to any before it
122  for (int i = 0; i < first; ++i)
123  {
124  m_io->m_cs[i] = m_io->m_cs[first];
125  }
126  int last = FindLastValidCrossSection();
127  // copy this cross section to any after it
128  for (size_t i = (size_t)(last + 1); last > 0 && i < m_io->m_cs.size(); ++i)
129  {
130  m_io->m_cs[i] = m_io->m_cs[last];
131  }
132  // loop between first and last to fill in the cross sections
133  const VecPt3d& pts(m_io->m_centerLine);
134  int prev = first, next;
135  double prevDist, nextDist, percent;
136  for (int i = first + 1; i < last; ++i)
137  {
138  auto& cs(m_io->m_cs[i]);
139  if (cs.m_left.size() > 1 || cs.m_right.size() > 1)
140  {
141  prev = i;
142  continue;
143  }
144 
145  // get the next valid cross section
146  next = FindNextValidCrossSection(i);
147  // get distance between current and previous
148  prevDist = gmXyDistance(pts[prev], pts[i]);
149  // get distance between current and next
150  nextDist = gmXyDistance(pts[i], pts[next]);
151  percent = prevDist / (prevDist + nextDist);
152  // interpolate the cross section
153  InterpCs(prev, next, percent, cs);
154  }
155 } // XmStampInterpCrossSectionImpl::InterpMissingCrossSections

244 void XmStampInterpCrossSectionImpl::InterpCs(XmStampCrossSection& a_prev,
245  XmStampCrossSection& a_next,
246  double a_percent,
247  XmStampCrossSection& a_cs)
248 {
249  XmStampCrossSection &pCs(a_prev), &nCs(a_next);
250  XM_ENSURE_TRUE(!pCs.m_left.empty() || !pCs.m_right.empty());
251  XM_ENSURE_TRUE(!nCs.m_left.empty() || !nCs.m_right.empty());
252  // get the interpolated center line location because the Interp pts
253  // method will not do the t = 0 location
254  Pt2d pt, p1, p2;
255  if (!pCs.m_left.empty())
256  p1 = pCs.m_left[0];
257  else if (!pCs.m_right.empty())
258  p1 = pCs.m_right[0];
259  if (!nCs.m_left.empty())
260  p2 = nCs.m_left[0];
261  else if (!nCs.m_right.empty())
262  p2 = nCs.m_right[0];
263  pt.x = p1.x - (a_percent * (p1.x - p2.x));
264  pt.y = p1.y - (a_percent * (p1.y - p2.y));
265  a_cs.m_left.assign(1, pt);
266  a_cs.m_right.assign(1, pt);
267 
268  // Left side interpolation
269  VecPt2d &l1(pCs.m_left), &l2(nCs.m_left), &l3(a_cs.m_left);
270  int ls1(pCs.m_idxLeftShoulder), ls2(nCs.m_idxLeftShoulder);
271  // interpolate points between the center line and the left shoulder
272  InterpPts(l1, 0, ls1, l2, 0, ls2, a_percent, l3);
273  a_cs.m_idxLeftShoulder = std::max(0, (int)l3.size() - 1);
274  // interpolate points between left shoulder and end
275  int lend1((int)pCs.m_left.size() - 1), lend2((int)nCs.m_left.size() - 1);
276  InterpPts(l1, ls1, lend1, l2, ls2, lend2, a_percent, l3);
277  // interpolate max dist
278  a_cs.m_leftMax = a_prev.m_leftMax - (a_percent * (a_prev.m_leftMax - a_next.m_leftMax));
279 
280  // Right side interpolation
281  VecPt2d &r1(pCs.m_right), &r2(nCs.m_right), &r3(a_cs.m_right);
282  int rs1(pCs.m_idxRightShoulder), rs2(nCs.m_idxRightShoulder);
283  // interpolate points between the center line and the left shoulder
284  InterpPts(r1, 0, rs1, r2, 0, rs2, a_percent, r3);
285  a_cs.m_idxRightShoulder = std::max(0, (int)r3.size() - 1);
286  // interpolate points between left shoulder and end
287  int rend1((int)pCs.m_right.size() - 1), rend2((int)nCs.m_right.size() - 1);
288  InterpPts(r1, rs1, rend1, r2, rs2, rend2, a_percent, r3);
289  // interpolate max dist
290  a_cs.m_rightMax = a_prev.m_rightMax - (a_percent * (a_prev.m_rightMax - a_next.m_rightMax));
291 } // XmStampInterpCrossSectionImpl::InterpCs

297 void XmStampInterpCrossSectionImpl::InterpFromCenterlineProfile(
298  XmStamperIo& a_io,
299  XmStamperCenterlineProfile& a_profile)
300 {
301  XM_ENSURE_TRUE(!a_profile.m_distance.empty());
302  XM_ENSURE_TRUE(a_profile.m_distance.size() == a_profile.m_elevation.size());
303  XM_ENSURE_TRUE(a_profile.m_distance.size() == a_profile.m_cs.size());
304  XM_ENSURE_TRUE(!a_io.m_centerLine.empty());
305 
306  // calculate the distance along the center line
307  VecDbl dist(a_io.m_centerLine.size(), 0);
308  VecPt3d& pts(a_io.m_centerLine);
309  for (size_t i = 1; i < pts.size(); ++i)
310  {
311  Pt3d &p0(pts[i - 1]), &p1(pts[i]);
312  dist[i] = Mdist(p0.x, p0.y, p1.x, p1.y);
313  dist[i] += dist[i - 1];
314  }
315 
316  // init the cross sections
317  a_io.m_cs.assign(a_io.m_centerLine.size(), XmStampCrossSection());
318 
319  VecDbl& d(a_profile.m_distance);
320  std::map<double, size_t> mapDistIdx;
321  for (size_t i = 0; i < d.size(); ++i)
322  {
323  mapDistIdx.insert(std::make_pair(d[i], i));
324  }
325 
326  auto itBegin = mapDistIdx.begin();
327  auto itEnd = mapDistIdx.end();
328  auto itLast = --mapDistIdx.end();
329  // interpolate between values in the profile
330  for (size_t i = 0; i < dist.size(); ++i)
331  {
332  // get closest profile point equal or after current point
333  auto it1 = mapDistIdx.lower_bound(dist[i]);
334  if (it1 == itEnd || (it1 == itLast && it1->first <= dist[i]))
335  { // centerline point after or at the last point in the profile
336  a_io.m_centerLine[i].z = a_profile.m_elevation.back();
337  a_io.m_cs[i] = a_profile.m_cs.back();
338  continue;
339  }
340  else if (it1->first >= dist[i] && it1 == itBegin)
341  { // centerline point before or at the first point in the profile
342  a_io.m_centerLine[i].z = a_profile.m_elevation.front();
343  a_io.m_cs[i] = a_profile.m_cs.front();
344  continue;
345  }
346  auto it0 = it1;
347  if (it0 == itBegin)
348  {
349  XM_ASSERT(0);
350  continue;
351  }
352  it0--;
353  size_t ix0(it0->second), ix1(it1->second);
354 
355  // get interp factor
356  double percent = (dist[i] - it0->first) / (it1->first - it0->first);
357  // interp elevation
358  VecDbl& elev(a_profile.m_elevation);
359  a_io.m_centerLine[i].z = elev[ix0] + percent * (elev[ix1] - elev[ix0]);
360  // interp cross section
361  XmStampCrossSection& prev(a_profile.m_cs[ix0]);
362  XmStampCrossSection& next(a_profile.m_cs[ix1]);
363  XmStampCrossSection& cs(a_io.m_cs[i]);
364  InterpCs(prev, next, percent, cs);
365  }
366 } // XmStampInterpCrossSectionImpl::InterpFromCenterlineProfile

381 //------------------------------------------------------------------------------
382 void XmStampInterpCrossSectionImpl::InterpPts(VecPt2d& a_v1,
383  int a_beg1,
384  int a_end1,
385  VecPt2d& a_v2,
386  int a_beg2,
387  int a_end2,
388  double a_percent,
389  VecPt2d& a_interpPts)
390 {
391  VecDbl t1, t2;
392  CalcTvals(a_v1, a_beg1, a_end1, t1);
393  CalcTvals(a_v2, a_beg2, a_end2, t2);
394  // get a union of t values
395  SetDbl tvals(t1.begin(), t1.end());
396  tvals.insert(t2.begin(), t2.end());
397 
398  Pt2d p1, p2, pt;
399  tvals.erase(0.0);
400  for (const auto& t : tvals)
401  {
402  p1 = PtFromT(a_v1, a_beg1, t1, t);
403  p2 = PtFromT(a_v2, a_beg2, t2, t);
404  pt.x = p1.x - (a_percent * (p1.x - p2.x));
405  pt.y = p1.y - (a_percent * (p1.y - p2.y));
406  a_interpPts.push_back(pt);
407  }

417 void XmStampInterpCrossSectionImpl::CalcTvals(VecPt2d& a_v, int a_beg, int a_end, VecDbl& a_t)
418 {
419  if (a_v.empty())
420  return;
421  a_t.assign(1, 0);
422  // calculate parametric t values for each x value
423  double diff = a_v[a_end].x - a_v[a_beg].x;
424  for (int i = a_beg + 1; i <= a_end; ++i)
425  {
426  double t = (a_v[i].x - a_v[a_beg].x) / diff;
427  a_t.push_back(t);
428  }

438 Pt2d XmStampInterpCrossSectionImpl::PtFromT(VecPt2d& a_v, int a_beg, VecDbl& a_t, double a_tval)
439 {
440  if (a_v.empty())
441  return Pt2d();
442  if (a_t.size() == 1 && a_t[0] == 0.0)
443  return a_v[0];
444 
445  Pt2d r, p1, p2;
446  double t1(-1), t2(-1);
447  int idx(-1);
448  // find where t is
449  for (size_t i = 1; idx < 0 && i < a_t.size(); ++i)
450  {
451  if (a_tval < a_t[i])
452  {
453  idx = static_cast<int>(i) - 1;
454  p1 = a_v[a_beg + idx];
455  p2 = a_v[a_beg + idx + 1];
456  t1 = a_t[idx];
457  t2 = a_t[idx + 1];
458  }
459  else if (a_tval == a_t[i])
460  {
461  idx = static_cast<int>(i);
462  p1 = a_v[a_beg + idx];
463  t1 = a_t[idx];
464  }
465  }
466  // interpolate to t
467  if (a_tval == t1)
468  return p1;
469  double localt = (a_tval - t1) / (t2 - t1);
470  r.x = p1.x + localt * (p2.x - p1.x);
471  r.y = p1.y + localt * (p2.y - p1.y);
472  return r;
473 } // XmStampInterpCrossSectionImpl::PtFromT

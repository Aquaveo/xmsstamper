61 static void iBuildStamperIo(const std::string& a_path, XmStamperIo& a_io)
62 {
63  std::string fname(a_path + "xmsng_StamperIo.txt");
64  std::ifstream is(fname.c_str());
65  XM_ENSURE_TRUE(is.is_open());
66  std::string card;
67  XM_ENSURE_TRUE(is >> card);
68  XM_ENSURE_TRUE(stEqualNoCase(card, "STAMPER_IO_VERSION_1"));
69  XM_ENSURE_TRUE(a_io.ReadFromFile(is));
70  // Test to make sure we are reading the XmStamperIo files correctly
71  std::ofstream outFile(a_path + "StamperIo_out.txt");
72  a_io.WriteToFile(outFile, "STAMPER_IO_VERSION_1");
73  outFile.close();
74  TS_ASSERT_TXT_FILES_EQUAL(a_path + "StamperIo_out.txt", fname);
75 } // iBuildStamperIo

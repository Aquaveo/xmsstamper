140 bool iInterpTinToRaster(const boost::shared_ptr<const TrTin> &a_tin, XmStampRaster &a_raster,
141  int a_stampingType = 2)
142 {
143  XM_ENSURE_TRUE(a_tin != nullptr, false);
144  BSHP<InterpLinear> interp = InterpLinear::New();
145  BSHP<VecPt3d> pts(new VecPt3d());
146  *pts = a_tin->Points();
147  BSHP<VecInt> tris(new VecInt());
148  *tris = a_tin->Triangles();
149  interp->SetPtsTris(pts, tris);
150  interp->SetExtrapVal(XM_NODATA);
151  int rasterSize = static_cast<int>(a_raster.m_vals.size());
152  for (int i = 0; i < rasterSize; ++i)
153  {
154  float val = interp->InterpToPt(Pt3d(a_raster.GetLocationFromCellIndex(i)));
155  if (!EQ_TOL(val, XM_NODATA, XM_ZERO_TOL))
156  {
157  // Take the stamp value if the raster has none, regardless of stamping type.
158  if (EQ_TOL(a_raster.m_vals[i], a_raster.m_noData, XM_ZERO_TOL))
159  {
160  a_raster.m_vals[i] = val;
161  }
162  else if (a_stampingType == 0) // Cut stamp, take the minimum
163  {
164  a_raster.m_vals[i] = std::min(double(val), a_raster.m_vals[i]);
165  }
166  else if (a_stampingType == 1) // Fill stamp, take the maximum
167  {
168  a_raster.m_vals[i] = std::max(double(val), a_raster.m_vals[i]);
169  }
170  else // Both, always stamp the feature object
171  {
172  a_raster.m_vals[i] = val;
173  }
174  }
175  }
176  return true;
177 } // iInterpTinToRaster

521 {
